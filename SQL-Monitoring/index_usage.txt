SELECT 
    OBJECT_NAME(s.object_id) AS ObjectName,
    i.name AS IndexName,
    i.index_id,
    user_seeks,
    user_scans,
    user_lookups,
    user_updates
FROM 
    sys.dm_db_index_usage_stats AS s
JOIN 
    sys.indexes AS i
ON 
    s.object_id = i.object_id AND s.index_id = i.index_id
WHERE 
    OBJECTPROPERTY(s.object_id, 'IsUserTable') = 1
ORDER BY 
    user_seeks DESC;
